CREATE TABLE Consultas (
    id_consulta SERIAL PRIMARY KEY,  -- Auto incremento de 6 d√≠gitos
    id_animal INT NOT NULL REFERENCES Animais(id_animal),
    id_veterinario CHAR(11) NOT NULL REFERENCES Veterinarios(id_veterinario),
